### 附录：

#### 编程用字体选择标准：

1. 所有字符等宽
2. 简洁、清晰、规范的字符形体
3. 支持ASCII码为128以上的扩展字符集
4. 空白字符(ASCII: 0x20)与其他字符等宽
5. '1'、'l'和'i'等三个字符易于区分
6. '0'、'o'和'O'等三个字符易于区分
7. 双引号、单引号的前后部分易于区分，最好是镜像对称的
8. 清晰的标点符号外形，尤其是大括符、圆括符和方括符

    //*******************************************************************
    // FILE: IvrPlatform.cs
    // AUTHOR: xuan
    // REMARKS:
    // NOTE: IVR平台工厂类
    // REVISION HISTORY:
    //   2008-4-26 xuan  建立
    //*******************************************************************
    using System;
    using System.Collections.Generic;
    using System.Text;
    using System.Configuration;

    using IVR.BL;

    namespace IVR
    {
        public class IvrPlatform
        {
            /// <summary>
            /// 通道应答类型
            /// </summary>
            public enum PLATFORM_TYPE
            {
                GENERAL,
                ENGLISH,
                PTM_CS,
                PTM_CS_ENG
            };

            #region 字段
            private static readonly string englishNo = ConfigurationManager.AppSettings["EnglishIVR"];   //英文查询号码
            private static readonly string ptmCsNo = ConfigurationManager.AppSettings["PtmCsIVR"];       //PTMCS查询号码
            private static readonly string ptmCsEngNo = ConfigurationManager.AppSettings["PtmCsEngIVR"]; //PTMCS英文查询号码
            private static readonly bool isOldPtm = ConfigurationManager.AppSettings["IsOldPtm"] == "1"; //PTM2005查询

            private static readonly string ptmBsAddr = ConfigurationManager.AppSettings["PtmBsAddr"];                   //PTM2005查询地址
            private static readonly int ptmBsPort = Convert.ToInt32(ConfigurationManager.AppSettings["PtmBsPort"]);     //PTM2005查询端口
            private static readonly string ptm2008Addr = ConfigurationManager.AppSettings["Ptm2008Addr"];               //PTM2008查询地址
            private static readonly int ptm2008Port = Convert.ToInt32(ConfigurationManager.AppSettings["Ptm2008Port"]); //PTM2008查询端口
            private static readonly string ptmCsAddr = ConfigurationManager.AppSettings["PtmCsAddr"];                   //PTMCS  查询地址
            private static readonly int ptmCsPort = Convert.ToInt32(ConfigurationManager.AppSettings["PtmCsPort"]);     //PTMCS  查询端口

            protected VoiceChannel voxChannel = null;  //语音文件定义类
            protected IvrAnswer ivrAnw = null;         //IVR应答流程
            protected QueryTask queryTask = null;      //防伪物流查询任务类

            #endregion

            #region 属性
            public VoiceChannel VoxChannel
            {
                get
                {
                    return this.voxChannel;
                }
            }

            public IvrAnswer IvrAnw
            {
                get
                {
                    return this.ivrAnw;
                }
            }

            public QueryTask QryTask
            {
                get
                {
                    return this.queryTask;
                }
            }
            #endregion

            #region 构造构析
            private IvrPlatform(VoiceChannel voxChannel, IvrAnswer ivrAnw, QueryTask queryTask)
            {
                this.voxChannel = voxChannel;
                this.ivrAnw = ivrAnw;
                this.queryTask = queryTask;
            }
            #endregion

            #region 静态方法
            /// <summary>
            /// 建立IVR平台关联类
            /// </summary>
            /// <param name="path">主路径</param>
            /// <param name="calledNo">被叫号码</param>
            /// <param name="chInfo">线程附加数据</param>
            /// <param name="displayCallBack">界面显示回调方法</param>
            /// <returns></returns>
            public static IvrPlatform CreateIvrPlatform(string path, string calledNo, ChInfo chInfo, DisplayCallBack displayCallBack)
            {
                VoiceChannel voxChannel = null;
                IvrAnswer ivrAnw = null;
                QueryTask queryTask = null;
                XLibrary.QueryServices.Interface.ICommonQuery commonQuery = null;

                switch (GetPlatformType(calledNo))
                {
                    case PLATFORM_TYPE.GENERAL:
                    voxChannel = new DefaultVoiceChannel(path, calledNo);
                    ivrAnw = new DefaultIvrAnswer(path, chInfo, displayCallBack);
                    if (isOldPtm)
                    {
                        commonQuery = new XLibrary.QueryServices.CommPDID();
                        commonQuery.ServicesAddr = ptmBsAddr;
                        commonQuery.ServicesPort = ptmBsPort;
                    }
                    else
                    {
                        commonQuery = new XLibrary.QueryServices.CommPTM2008();
                        commonQuery.ServicesAddr = ptm2008Addr;
                        commonQuery.ServicesPort = ptm2008Port;
                    }
                    break;
                    case PLATFORM_TYPE.ENGLISH:
                    voxChannel = new EnglishVoiceChannel(path, calledNo);
                    ivrAnw = new EnglishIvrAnswer(path, chInfo, displayCallBack);
                    if (isOldPtm)
                    {
                        commonQuery = new XLibrary.QueryServices.CommPDID();
                        commonQuery.ServicesAddr = ptmBsAddr;
                        commonQuery.ServicesPort = ptmBsPort;
                    }
                    else
                    {
                        commonQuery = new XLibrary.QueryServices.CommPTM2008();
                        commonQuery.ServicesAddr = ptm2008Addr;
                        commonQuery.ServicesPort = ptm2008Port;
                    }
                    break;
                    case PLATFORM_TYPE.PTM_CS:
                    voxChannel = new PtmCsVoiceChannel(path, calledNo);
                    ivrAnw = new PtmCsIvrAnswer(path, chInfo, displayCallBack);
                    commonQuery = new XLibrary.QueryServices.CommPTMCS();
                    commonQuery.ServicesAddr = ptmCsAddr;
                    commonQuery.ServicesPort = ptmCsPort;
                    break;
                    case PLATFORM_TYPE.PTM_CS_ENG:
                    voxChannel = new PtmCsEngVoiceChannel(path, calledNo);
                    ivrAnw = new PtmCsEngIvrAnswer(path, chInfo, displayCallBack);
                    commonQuery = new XLibrary.QueryServices.CommPTMCS();
                    commonQuery.ServicesAddr = ptmCsAddr;
                    commonQuery.ServicesPort = ptmCsPort;
                    break;
                }
                queryTask = new QueryTask(commonQuery);
                return new IvrPlatform(voxChannel, ivrAnw, queryTask);
            }

            /// <summary>
            /// 得到通道应答类型
            /// </summary>
            /// <param name="calledNo">被叫号码</param>
            /// <returns></returns>
            public static PLATFORM_TYPE GetPlatformType(string calledNo)
            {
                if (englishNo.IndexOf(calledNo) >= 0)
                {
                    return PLATFORM_TYPE.ENGLISH;
                }
                else if (ptmCsNo.IndexOf(calledNo) >= 0)
                {
                    return PLATFORM_TYPE.PTM_CS;
                }
                else if (ptmCsEngNo.IndexOf(calledNo) >= 0)
                {
                    return PLATFORM_TYPE.PTM_CS_ENG;
                }

                return PLATFORM_TYPE.GENERAL;
            }

            #endregion
        }
    }
